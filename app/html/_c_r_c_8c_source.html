20 *
21 * @param DesBuf 输出缓冲区，存储反转后的字节
22 * @param SrcBuf 输入缓冲区，包含要反转的字节
23 *
24 * @note 此函数将输入字节的位顺序反转并存储到输出缓冲区。
25 */
26void InvertUint8(uint8_t *DesBuf, uint8_t *SrcBuf)
27{
28 int i;
29 uint8_t temp = 0;
30
31 for (i = 0; i < 8; i++)
32 {
33 if (SrcBuf[0] & (1 << i))
34 {
35 temp |= 1 << (7 - i);
36 }
37 }
38 DesBuf[0] = temp;
39}

62/**
63 * @brief 计算 CRC-16/X.25 校验码
64 *
65 * @param puchMsg 指向输入数据的指针
66 * @param usDataLen 输入数据的长度
67 * @return uint16_t 计算得到的 CRC 校验码
68 *
69 * @note 此函数实现 CRC-16/X25 校验算法，输入数据反转后计算 CRC 值。
70 */
71uint16_t CRC16_X25(uint8_t *puchMsg, unsigned int usDataLen)
72{
73 uint16_t wCRCin = 0xFFFF;
74 uint16_t wCPoly = 0x1021;
75 uint8_t wChar = 0;
76
77 while (usDataLen--)
78 {
79 wChar = *(puchMsg++);
80 InvertUint8(&wChar, &wChar);
81 wCRCin ^= (wChar << 8);
82
83 for (int i = 0; i < 8; i++)
84 {
85 if (wCRCin & 0x8000)
86 {
87 wCRCin = (wCRCin << 1) ^ wCPoly;
88 }
89 else
90 {
91 wCRCin = wCRCin << 1;
92 }
93 }
94 }
95 InvertUint16(&wCRCin, &wCRCin);
96 return (wCRCin ^ 0xFFFF);
97}
