102uint8_t WT901C_SumCRC(uint8_t Type, uint8_t *data)
103{
104 uint8_t i;
105 uint8_t Sum = WT901C_HEAD + Type;
106 for (i = 0; i < 8; i++)
107 {
108 Sum += data[i];
109 }
110 return Sum;
111}
