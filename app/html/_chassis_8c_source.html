93void Chassis_Task(void const *argument)
94{

103 /* 陀螺仪串口接收中断 */
104 HAL_UART_Receive_IT(&huart2, &Rx_buf, 1);
105 // PID_Init(&chassis.Angle_pid, 2, 1, 1, 100, 100);//角度环
106 // chassis.set_angle = IMU_angle.Yaw; // 角度环pid

109 PID_Init(&chassis.Motor2.pid, 5000, 10, 100, 1000, 1000);
110 PID_Init(&chassis.Motor3.pid, 5000, 10, 100, 1000, 1000);
111 PID_Init(&chassis.Motor4.pid, 5000, 10, 100, 1000, 1000);
112 control.Vx = control.Vy = control.Vz = 0;
113 while (1)
114 {
115

125 /* 计算设置速度 */
126 Motor_Speed(control.Vx, control.Vy, control.Vz);

128 /* 计算实际速度 */
129 time_count = 0;
130 chassis.Motor1.Encode_Count = chassis.Motor2.Encode_Count = chassis.Motor3.Encode_Count = chassis.Motor4.Encode_Count = 0;
131 while (time_count < 10)
132 ;
133 count_buf1 = chassis.Motor1.Encode_Count;
134 count_buf2 = chassis.Motor2.Encode_Count;
135 count_buf3 = chassis.Motor3.Encode_Count;
136 count_buf4 = chassis.Motor4.Encode_Count;
137
138 chassis.Motor1.rps = count_buf1 / Pulse_Num * 100;
139 chassis.Motor1.speed = chassis.Motor1.rps * Motor_PI * Wheel_Diameter * chassis.Motor1.dir;
140
141 chassis.Motor2.rps = count_buf2 / Pulse_Num * 100;
142 chassis.Motor2.speed = chassis.Motor2.rps * Motor_PI * Wheel_Diameter * chassis.Motor2.dir;
143
144 chassis.Motor3.rps = count_buf3 / Pulse_Num * 100;
145 chassis.Motor3.speed = chassis.Motor3.rps * Motor_PI * Wheel_Diameter * chassis.Motor3.dir;
146
147 chassis.Motor4.rps = count_buf4 / Pulse_Num * 100;
148 chassis.Motor4.speed = chassis.Motor4.rps * Motor_PI * Wheel_Diameter * chassis.Motor4.dir;
149
150 /* PID计算 */
151 PID_Calc(&chassis.Motor1.pid, chassis.Motor1.Set_speed * Speed_Factor, chassis.Motor1.speed);
152 PID_Calc(&chassis.Motor2.pid, chassis.Motor2.Set_speed * Speed_Factor, chassis.Motor2.speed);
153 PID_Calc(&chassis.Motor3.pid, chassis.Motor3.Set_speed * Speed_Factor, chassis.Motor3.speed);
154 PID_Calc(&chassis.Motor4.pid, chassis.Motor4.Set_speed * Speed_Factor, chassis.Motor4.speed);
155 Motor_PWM();
156 }
157}

300/**
301 * @brief 计算设置速度
302 *
303 * @param vx X轴方向速度
304 * @param vy Y轴方向速度
305 * @param RotateV 旋转速度，顺时针为负值，逆时针为正值
306 */
307void Motor_Speed(int16_t vx, int16_t vy, int16_t RotateV)
308{
309 vx = vx / 4;
310 vy = vy / 4;
311 RotateV /= 4;
312
313 chassis.Motor1.Set_speed = -vy + -vx + -RotateV;
314 chassis.Motor2.Set_speed = -vy + vx + -RotateV;
315 chassis.Motor3.Set_speed = vy + vx + -RotateV;
316 chassis.Motor4.Set_speed = vy + -vx + -RotateV;
317
318 Limit_Max(chassis.Motor1.Set_speed, MaxSpeed);
319 Limit_Max(chassis.Motor2.Set_speed, MaxSpeed);
320 Limit_Max(chassis.Motor3.Set_speed, MaxSpeed);
321 Limit_Max(chassis.Motor4.Set_speed, MaxSpeed);
322}
323/**
324 * @brief 底盘串口回调函数
325 * @note 串口接收底盘运动控制信息
326 *
327 */
328void Chassis_UART_RxCpltCallback(void)
329{
330 if (Usart_Head == Packet_Head && Usart_Flag == 0)
331 {
332 Usart_Flag = 1;
333 HAL_UART_Receive_IT(&huart1, data, 9);
334 return;
335 }
336 if (Usart_Flag == 1 && data[8] == Packet_End)
337 {
338 Usart_Flag = 0;
339 CRC16 = CRC16_X25(&data[0], 6);
340 if (CRC16 / 256 == data[6] & CRC16 % 256 == data[7])
341 {
342 control.Vx = data[0] << 8 | data[1]; // x
343 control.Vy = data[2] << 8 | data[3]; // y
344 control.Vz = data[4] << 8 | data[5]; // rotate
345 Data_Flag = 0;
346 }
347 else
348 {
349 Data_Flag = 1;
350 }
351 }
352 else if (Usart_Flag)
353 {
354 Usart_Flag = 0;
355 Data_Flag = 2;
356 }
357 HAL_UART_Receive_IT(&huart1, &Usart_Head, 1);
358}
